def calculate_technical_indicators(ticker_symbol, period="1y"):
    """ä¸»è¦ãªãƒ†ã‚¯ãƒ‹ã‚«ãƒ«æŒ‡æ¨™ã‚’è¨ˆç®—"""
    ticker = yf.Ticker(ticker_symbol)
    df = ticker.history(period=period)
    
    # ç§»å‹•å¹³å‡
    df['SMA_20'] = df['Close'].rolling(window=20).mean()
    df['SMA_50'] = df['Close'].rolling(window=50).mean()
    df['SMA_200'] = df['Close'].rolling(window=200).mean()
    
    # æŒ‡æ•°ç§»å‹•å¹³å‡ï¼ˆEMAï¼‰
    df['EMA_12'] = df['Close'].ewm(span=12, adjust=False).mean()
    df['EMA_26'] = df['Close'].ewm(span=26, adjust=False).mean()
    
    # MACD
    df['MACD'] = df['EMA_12'] - df['EMA_26']
    df['Signal'] = df['MACD'].ewm(span=9, adjust=False).mean()
    
    # RSI (Relative Strength Index)
    delta = df['Close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # ãƒœãƒªãƒ³ã‚¸ãƒ£ãƒ¼ãƒãƒ³ãƒ‰
    df['BB_Middle'] = df['Close'].rolling(window=20).mean()
    std = df['Close'].rolling(window=20).std()
    df['BB_Upper'] = df['BB_Middle'] + (std * 2)
    df['BB_Lower'] = df['BB_Middle'] - (std * 2)
    
    return df

# ä½¿ç”¨ä¾‹
df = calculate_technical_indicators("AAPL")

def calculate_risk_metrics(ticker_symbol, period="1y"):
    """ãƒªã‚¹ã‚¯é–¢é€£æŒ‡æ¨™ã‚’è¨ˆç®—"""
    ticker = yf.Ticker(ticker_symbol)
    df = ticker.history(period=period)
    
    # æ—¥æ¬¡ãƒªã‚¿ãƒ¼ãƒ³
    daily_returns = df['Close'].pct_change()
    
    # å¹´ç‡ãƒªã‚¿ãƒ¼ãƒ³
    annual_return = daily_returns.mean() * 252
    
    # ãƒœãƒ©ãƒ†ã‚£ãƒªãƒ†ã‚£ï¼ˆå¹´ç‡ï¼‰
    volatility = daily_returns.std() * np.sqrt(252)
    
    # ã‚·ãƒ£ãƒ¼ãƒ—ãƒ¬ã‚·ã‚ªï¼ˆãƒªã‚¹ã‚¯ãƒ•ãƒªãƒ¼ãƒ¬ãƒ¼ãƒˆ2%ã¨ä»®å®šï¼‰
    risk_free_rate = 0.02
    sharpe_ratio = (annual_return - risk_free_rate) / volatility
    
    # æœ€å¤§ãƒ‰ãƒ­ãƒ¼ãƒ€ã‚¦ãƒ³
    cumulative = (1 + daily_returns).cumprod()
    running_max = cumulative.expanding().max()
    drawdown = (cumulative - running_max) / running_max
    max_drawdown = drawdown.min()
    
    # VaR (Value at Risk) - 95%ä¿¡é ¼åŒºé–“
    var_95 = np.percentile(daily_returns.dropna(), 5)
    
    return {
        'annual_return': annual_return,
        'volatility': volatility,
        'sharpe_ratio': sharpe_ratio,
        'max_drawdown': max_drawdown,
        'var_95': var_95
    }

# ä½¿ç”¨ä¾‹

def create_technical_dashboard(ticker_symbol, period="1y"):
    """ãƒ†ã‚¯ãƒ‹ã‚«ãƒ«åˆ†æãƒ€ãƒƒã‚·ãƒ¥ãƒœãƒ¼ãƒ‰"""
    ticker = yf.Ticker(ticker_symbol)
    df = ticker.history(period=period)
    
    # ãƒ†ã‚¯ãƒ‹ã‚«ãƒ«æŒ‡æ¨™ã‚’è¨ˆç®—
    df['SMA_20'] = df['Close'].rolling(window=20).mean()
    df['SMA_50'] = df['Close'].rolling(window=50).mean()
    
    # RSI
    delta = df['Close'].diff()
    gain = delta.where(delta > 0, 0).rolling(window=14).mean()
    loss = -delta.where(delta < 0, 0).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # MACD
    exp1 = df['Close'].ewm(span=12, adjust=False).mean()
    exp2 = df['Close'].ewm(span=26, adjust=False).mean()
    df['MACD'] = exp1 - exp2
    df['Signal'] = df['MACD'].ewm(span=9, adjust=False).mean()
    
    # ã‚µãƒ–ãƒ—ãƒ­ãƒƒãƒˆä½œæˆ
    fig = make_subplots(
        rows=3, cols=1,
        subplot_titles=['æ ªä¾¡ & ç§»å‹•å¹³å‡', 'RSI', 'MACD'],
        vertical_spacing=0.08,
        row_heights=[0.5, 0.25, 0.25]
    )
    
    # 1. ãƒ­ãƒ¼ã‚½ã‚¯è¶³ & ç§»å‹•å¹³å‡
    fig.add_trace(go.Candlestick(
        x=df.index,
        open=df['Open'],
        high=df['High'],
        low=df['Low'],
        close=df['Close'],
        name='OHLC'
    ), row=1, col=1)
    
    fig.add_trace(go.Scatter(
        x=df.index, y=df['SMA_20'],
        name='SMA 20', line=dict(color='orange', width=1)
    ), row=1, col=1)
    
    fig.add_trace(go.Scatter(
        x=df.index, y=df['SMA_50'],

    ), row=1, col=1)
    
    # 2. RSI
    fig.add_trace(go.Scatter(
        x=df.index, y=df['RSI'],
        name='RSI', line=dict(color='purple')
    ), row=2, col=1)
    
    # RSIã®70ã¨30ã®ãƒ©ã‚¤ãƒ³
    fig.add_hline(y=70, line_dash="dash", line_color="red",
                  annotation_text="Overbought", row=2, col=1)

                  annotation_text="Oversold", row=2, col=1)
    
    # 3. MACD
    fig.add_trace(go.Scatter(
        x=df.index, y=df['MACD'],
        name='MACD', line=dict(color='blue')
    ), row=3, col=1)
    
    fig.add_trace(go.Scatter(
        x=df.index, y=df['Signal'],
        name='Signal', line=dict(color='orange')
    ), row=3, col=1)

    fig.update_yaxes(title_text="RSI", row=2, col=1)
